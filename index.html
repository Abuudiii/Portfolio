Project 1: Autoirrigator

Short Description: The Autoirrigator is an automated irrigation system that measures soil moisture levels and waters plants accordingly. It uses a keypad for user input and an RGB LED to visually indicate soil moisture levels.

Arduino Code

            
// Autoirrigator Arduino Code

#include <Keypad.h>

const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};

byte rowPins[ROWS] = {9, 8, 7, 6};
byte colPins[COLS] = {5, 4, 3, 2};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

int waterPumpPin = 10;
int moistureSensorPin = A1;
const int wet = 484;
const int dry = 898;
const int flowRate = 100;
int targetWaterVolume = 0;
int maxContainerVolume = 1000;
int currentWaterLevel = 1000;
const int moistureThreshold = 70;

int redPin = 13;
int greenPin = 11;
int bluePin = 12;

void setup() {
  SerialUSB.begin(9600);
  pinMode(waterPumpPin, OUTPUT);
  digitalWrite(waterPumpPin, LOW);
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
}

void loop() {
  int moistureLevel = analogRead(moistureSensorPin);
  int percentageMoisture = map(moistureLevel, wet, dry, 100, 0);

  SerialUSB.print("Soil Moisture: ");
  SerialUSB.print(percentageMoisture);
  SerialUSB.println("%");

  if (percentageMoisture <= 20) {
    digitalWrite(redPin, HIGH);
    digitalWrite(greenPin, LOW);
    digitalWrite(bluePin, LOW);
  } else if (percentageMoisture <= 60) {
    digitalWrite(redPin, HIGH);
    digitalWrite(greenPin, HIGH);
    digitalWrite(bluePin, LOW);
  } else {
    digitalWrite(redPin, LOW);
    digitalWrite(greenPin, HIGH);
    digitalWrite(bluePin, LOW);
  }

  char key = keypad.getKey();
  if (key) {
    if (key == '*') {
      targetWaterVolume = 0;
      SerialUSB.println("Reset input.");
    } else if (key == '#') {
      if (percentageMoisture > moistureThreshold) {
        SerialUSB.println("Moisture level is sufficient, no watering needed.");
      } else if (targetWaterVolume > currentWaterLevel) {
        SerialUSB.println("Not enough water in the container!");
      } else {
        SerialUSB.print("Dispensing ");
        SerialUSB.print(targetWaterVolume);
        SerialUSB.println(" mL of water.");
        dispenseWater(targetWaterVolume);
        currentWaterLevel -= targetWaterVolume;
        SerialUSB.print("Remaining water: ");
        SerialUSB.print(currentWaterLevel);
        SerialUSB.println(" mL");
      }
    } else if (isDigit(key)) {
      targetWaterVolume = targetWaterVolume * 10 + (key - '0');
      SerialUSB.print("Current input: ");
      SerialUSB.print(targetWaterVolume);
      SerialUSB.println(" mL");
      if (targetWaterVolume > maxContainerVolume) {
        targetWaterVolume = maxContainerVolume;
        SerialUSB.println("Max volume is 1000 mL.");
      }
    }
  }
  delay(100);
}

void dispenseWater(int volume) {
  if (volume > 0 && volume <= 1000) {
    int dispenseTime = (volume * 1000) / flowRate;
    digitalWrite(waterPumpPin, HIGH);
    delay(dispenseTime);
    digitalWrite(waterPumpPin, LOW);
    SerialUSB.println("Dispensing complete.");
  } else {
    SerialUSB.println("Invalid water volume.");
  }
}
