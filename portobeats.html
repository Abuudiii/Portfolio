Project Overview

Portobeats is a wearable glove system that allows users to assign different musical instruments to each finger tap on any surface, enabling them to create beats interactively. This project combines wearable technology and music production to create a tangible interaction experience that encourages creativity and exploration.

Interaction Details

Each glove is equipped with force-sensing resistors that can detect finger taps on any surface. The system is connected to a MIDI controller, allowing users to assign specific instruments or drum sounds to each finger. When a user taps a surface with a particular finger, the corresponding sound is produced, creating a musical beat. This makes it possible to generate dynamic and engaging rhythms using natural hand movements.

Project Code


#include <MIDI.h>
#include <CapacitiveSensor.h>

// Define pin connections for capacitive sensors
CapacitiveSensor cs1 = CapacitiveSensor(4, 2); // Thumb
CapacitiveSensor cs2 = CapacitiveSensor(4, 3); // Index finger
CapacitiveSensor cs3 = CapacitiveSensor(4, 5); // Middle finger
CapacitiveSensor cs4 = CapacitiveSensor(4, 6); // Ring finger
CapacitiveSensor cs5 = CapacitiveSensor(4, 7); // Pinky finger

MIDI_CREATE_DEFAULT_INSTANCE();

void setup() {
    MIDI.begin(MIDI_CHANNEL_OMNI);
    Serial.begin(9600);
}

void loop() {
    long thumbValue = cs1.capacitiveSensor(30);
    long indexValue = cs2.capacitiveSensor(30);
    long middleValue = cs3.capacitiveSensor(30);
    long ringValue = cs4.capacitiveSensor(30);
    long pinkyValue = cs5.capacitiveSensor(30);

    if (thumbValue > 1000) {
        MIDI.sendNoteOn(36, 127, 1); // Kick drum
        delay(100);
        MIDI.sendNoteOff(36, 0, 1);
    }
    if (indexValue > 1000) {
        MIDI.sendNoteOn(38, 127, 1); // Snare drum
        delay(100);
        MIDI.sendNoteOff(38, 0, 1);
    }
    if (middleValue > 1000) {
        MIDI.sendNoteOn(42, 127, 1); // Hi-hat
        delay(100);
        MIDI.sendNoteOff(42, 0, 1);
    }
    if (ringValue > 1000) {
        MIDI.sendNoteOn(46, 127, 1); // Open hi-hat
        delay(100);
        MIDI.sendNoteOff(46, 0, 1);
    }
    if (pinkyValue > 1000) {
        MIDI.sendNoteOn(49, 127, 1); // Crash cymbal
        delay(100);
        MIDI.sendNoteOff(49, 0, 1);
    }
}
